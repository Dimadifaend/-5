
#include <stdio.h>
#include <locale.h>
#include <math.h>
int main()
{
	int x, y, x1, y1;

	setlocale(LC_ALL, "RUS");

	scanf("%d", &y);
	x1 = x % 2;
	y1 = y % 2;
	if (x1 == 0 && y1!=0)

		printf("Первый ход должена сделать команада А\n");
	}

		printf("Первый ход должена сделать команада B\n");
	}
	else 
	{
		printf("Закройте окно и повторите попытку, введите числа так, чтобы одно число было чётное, а другое нет\n");
